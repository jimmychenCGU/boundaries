    <script>

        function calculateScore() {

            const form = document.getElementById('quizForm');

            const questions = ['q1', 'q2', 'q3', 'q4', 'q5'];

            let yesCount = 0;

            let answers = [];

 

            questions.forEach(q => {

                const radio = form.querySelector(`input[name="${q}"]:checked`);

                const feel = form.querySelector(`textarea[name="${q}-feel"]`).value;

                if (radio && radio.value === 'yes') {

                    yesCount++;

                }

                answers.push({

                    q: q,

                    answer: radio ? radio.value : '未選',

                    feel: feel

                });

            });

 

            const resultDiv = document.getElementById('result');

            let html = `<h3>你的結果：</h3><p class="yes-count">你有 <strong>${yesCount}</strong> 個「是」答案！</p>`;

 

            if (yesCount >= 3) {

                html += '<p style="color: green;">🎉 恭喜！你的邊界相對健康。你已經能有效保護自己的空間，讓關係更平衡。繼續保持，並分享給朋友練習吧！</p>';

            } else {

                html += '<p style="color: orange;">💡 你的邊界可能有不足傾向，別擔心，這是成長機會！以下是專屬建議：</p>';

                html += '<div class="suggestion">';

                html += '<ul>';

                html += '<li><strong>學習面對不適：</strong> 從小事練習說「不」，如用「我了解你的需求，但現在我無法幫忙」開頭，配上微笑。這能減輕內疚感，找回自主。</li>';

                html += '<li><strong>建立日常邊界：</strong> 每天預留「我時間」（如散步或冥想），提醒自己：你的感受同樣重要，不是自私。</li>';

                html += '<li><strong>尋求支持：</strong> 找信任的朋友或專業諮詢（如Jimmy醫師），分享你的掙扎。他們能幫你驗證邊界不是拒絕愛，而是讓關係更平衡。</li>';

                html += '<li><strong>閱讀延伸：</strong> 重溫《Boundaries》一書的順從型章節，試著記錄一週內的「妥協時刻」，並反思「如果我說不，會發生什麼？」</li>';

                html += '</ul></div>';

            }

 

            html += '<p><em>記住，設邊界是成長的禮物，讓你更自由地愛自己和他人。如果你想分享你的測試結果，我們可以一起討論！</em></p>';

            html += '<button onclick="resetForm()">重新測試</button>';

 

            resultDiv.innerHTML = html;

            resultDiv.style.display = 'block';

        }

 

        function resetForm() {

            document.getElementById('quizForm').reset();

            document.getElementById('result').style.display = 'none';

        }

    </script>
